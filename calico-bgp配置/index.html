Calico Bgp配置

calico-bgp配置 #

calicoctl 配置 bgp #

bgp

# 查看当前node节点的bgp配置
calicoctl node status

# 查看ASN号
calicoctl get nodes -o wide

# 关闭fullmesh
calicoctl patch bgpconfiguration default --type='merge' -p '{"spec":{"nodeToNodeMesh":false}}'

# 修改节点的calico配置
calicoctl get nodes -o yaml > nodes.yaml
# 编辑 nodes.yaml 添加 asn(后面配置的rrgroup) 和 rr的ip(就是自己的节点ip)

# 先给node打标签，区分rrgroupid和rr节点
kubectl label node k8s-m1 bgp_id=65040  # 充当rr节点
kubectl label node k8s-m1 bgp_group=65040 # 同一个asn的节点

# 配置bgppeer 和 邻居leaf01交换机 **EBGP**
apiVersion: projectcalico.org/v3
kind: BGPPeer
metadata:
  name: rr65040-to-leaf01
spec:
  nodeSelector: "bgp_id == '65040'"
  peerIP: # 邻居leaf01交换机leaf01的ip地址
  asNumber: 650009 # 邻居leaf01交换机的asNumber，不是自己的

# 配置bgppeer 和 其他rr节点 **IBGP**
apiVersion: projectcalico.org/v3
kind: BGPPeer
metadata:
  name: rr65040-to-node-peers
spec:
  nodeSelector: "bgp_group == '65040'"
  peerSelector: "bgp_id == '65040'"

calicoctl apply -f bgppeer-rr-to-leaf01.yaml
calicoctl apply -f bgppeer-rr-to-node-peers.yaml


# 查看bgppeer
calicoctl get bgppeer -o wide
calicoctl node status

# leaf01交换机 查看bgp
show bgp peer ipv4
show bgp routing-table ipv4

# leaf01交换机 设置bgp
router bgp 65001  # 交换机的 AS 号
  peer K8S_NODES group
  peer K8S_NODES remote-as 65002  # 假设 Node 是另一个 AS
  peer 192.168.1.10 group K8S_NODES # 指向具体的 K8s Node IP
  address-family ipv4 unicast
    peer K8S_NODES enable
