1.3. yaml准备 #

apiVersion: v1
kind: Namespace
metadata:
  name: comfyui
---
apiVersion: apps/v1
kind: StatefulSet
metadata:
  name: comfyui-sts
  namespace: comfyui
spec:
  updateStrategy:
    type: OnDelete
  serviceName: comfyui-service
  replicas: 1
  selector:
    matchLabels:
      app: comfyui
  template:
    metadata:
      labels:
        app: comfyui
    spec:
      # 必须指定调度到有 GPU 的节点
      nodeSelector:
        kubernetes.io/hostname: "prod-m1"
      # 使用我们在 containerd 中配置的 nvidia 运行时
      runtimeClassName: nvidia 
      containers:
      - name: comfyui
        # image: yanwk/comfyui-boot:cu128-megapak-20260221
        image: yanwk/comfyui-boot:cu126-slim-20260223
        ports:
        - containerPort: 8188
        env:
        # disable-xformers 选项会禁用 xformers 库的使用，强制 ComfyUI 使用标准的 PyTorch 实现，更稳定。
        - name: CLI_ARGS
          value: "--listen 0.0.0.0 --disable-xformers"
        resources:
          limits:
            nvidia.com/gpu: 1           # 占用 1 个 Hami 虚拟单位
            # nvidia.com/gpumem: 4000     # 限制 4GB 显存，这会触发 Hami 的隔离
            # nvidia.com/gpucores: 50     # 限制 50% 算力（可选）
        volumeMounts:
        - name: storage
          mountPath: /root
        - name: models
          mountPath: /root/ComfyUI/models
        - name: hf-hub
          mountPath: /root/.cache/huggingface/hub
        - name: torch-hub
          mountPath: /root/.cache/torch/hub
        - name: input
          mountPath: /root/ComfyUI/input
        - name: output
          mountPath: /root/ComfyUI/output
        - name: workflows
          mountPath: /root/ComfyUI/user/default/workflows
      volumes:
      # 这里将你本地的目录映射进去，请将 /path/to/ 修改为你的实际绝对路径
      - name: storage
        hostPath: { path: /data/comfyui/storage }
      - name: models
        hostPath: { path: /data/comfyui/storage-models/models }
      - name: hf-hub
        hostPath: { path: /data/comfyui/storage-models/hf-hub }
      - name: torch-hub
        hostPath: { path: /data/comfyui/storage-models/torch-hub }
      - name: workflows
        hostPath: { path: /data/comfyui/storage-user/workflows }
  # input/output 使用 NFS SC 动态创建 PVC，支持扩容（allowVolumeExpansion: true）
  volumeClaimTemplates:
  - metadata:
      name: input
      annotations:
        nfs.io/storage-path: comfyui-input
    spec:
      accessModes: ["ReadWriteMany"]
      storageClassName: annotations-path-retain
      resources:
        requests:
          storage: 20Gi
  - metadata:
      name: output
      annotations:
        nfs.io/storage-path: comfyui-output
    spec:
      accessModes: ["ReadWriteMany"]
      storageClassName: annotations-path-retain
      resources:
        requests:
          storage: 50Gi
---
apiVersion: v1
kind: Service
metadata:
  name: comfyui-service
  namespace: comfyui
spec:
  type: NodePort
  selector:
    app: comfyui
  ports:
    - protocol: TCP
      port: 8188
      targetPort: 8188
      nodePort: 30188

# 工作流

3. 基于 HAMi 的部署要点 #

  • Pod/StatefulSet 指定 runtimeClassName: nvidia
  • 资源限制声明 nvidia.com/gpu: 1 好像申请多了会一直pending
  • 将模型与缓存目录通过 hostPath 或 PVC 挂载

4. 基于 GPU Operator 的部署要点 #

  • resources.limits 必须包含 nvidia.com/gpu
  • 节点需能看到 nvidia.com/gpu: 1 等标签/资源

5. 最小资源片段 #

resources:
  limits:
    nvidia.com/gpu: 1
