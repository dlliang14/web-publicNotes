Js开发 Bom与dom

js开发-bom与dom #

js开发

BOM（浏览器对象模型） #

window 可以省略，直接调用 history、location、document 等对象。

History #

history.back();      // 后退
history.forward();   // 前进
history.go(-1);      // 后退
history.go(1);       // 前进
history.go(0);       // 刷新当前页面

Location #

location.href;       // 获取或设置 URL
location.protocol;   // 获取或设置协议
location.host;       // 获取或设置主机
location.reload();   // 刷新页面

通知弹窗 #

alert('这是一个警告框');
confirm('你确定要删除吗？');
prompt('请输入你的名字：');

DOM（文档对象模型） #

常用属性 #

document.title; // 获取或设置页面标题

常用选择器 #

document.getElementById('id');
document.getElementsByClassName('class'); // HTMLCollection
document.getElementsByTagName('tag');     // HTMLCollection
document.querySelector('selector');
document.querySelectorAll('selector');    // NodeList

节点关系 #

// 包含文本节点
element.childNodes;
element.firstChild;
element.lastChild;
element.parentNode;
element.previousSibling;
element.nextSibling;

// 仅元素节点
element.children;
element.firstElementChild;
element.lastElementChild;
element.parentElement;
element.previousElementSibling;
element.nextElementSibling;

新增节点 #

const newElement = document.createElement('div');
newElement.textContent = 'Hello, World!';
const container = document.getElementById('container');
container.appendChild(newElement);
container.insertBefore(newElement, container.firstChild);

删除节点 #

const elementToRemove = document.getElementById('element-id');
elementToRemove.parentNode.removeChild(elementToRemove);
// 现代浏览器
// elementToRemove.remove();

修改内容与属性 #

const target = document.querySelector('.video-share-info-text');
target.textContent = '888';

const shareInfo = document.querySelector('.video-share-info');
shareInfo.setAttribute('data-shared', 'true');
shareInfo.getAttribute('data-shared');
shareInfo.removeAttribute('data-shared');

shareInfo.classList.add('active');
shareInfo.classList.remove('active');
shareInfo.classList.toggle('active');
shareInfo.classList.contains('active');

shareInfo.style.color = 'red';
shareInfo.style.fontSize = '16px';
