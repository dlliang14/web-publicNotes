删除失败的补救 #

  • 忘记指定名称空间，想要删除argocd
  • 但是一直卡在terminating状态
# 查看资源
for api in $(kubectl api-resources --verbs=list --namespaced -o name); do
  kubectl get  $api --ignore-not-found
done

# 强制删除指定阶段的资源 默认名称空间 Terminating
kubectl get pods  | grep Terminating | awk '{print $1}' \
  | xargs -r -n1 kubectl delete pod  --force --grace-period=0

argo路由 #

  • 强制https
# 使用nginx的ingress路由
# 域名选择 argocd.180007.xyz
cat > argocd-ingress.yaml <<EOF
# argocd-ingress.yaml
apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: argocd-ingress
  namespace: argocd
  annotations:
    kubernetes.io/ingress.class: "nginx"
    nginx.ingress.kubernetes.io/force-ssl-redirect: "true"
    nginx.ingress.kubernetes.io/backend-protocol: "HTTP"
    cert-manager.io/cluster-issuer: "cloudflare-prod-issue"
spec:
  ingressClassName: nginx
  rules:
  - host: argocd.180007.xyz
    http:
      paths:
      - path: /
        pathType: Prefix
        backend:
          service:
            name: argocd-server
            port:
              number: 80
  tls:
  - hosts:
    - argocd.180007.xyz
    secretName: argocd-tls-secret
EOF
kubectl apply -f argocd-ingress.yaml

# grpc访问
cat > argocd-ingress-grpc.yaml <<EOF
# argocd-ingress-grpc.yaml
apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: argocd-ingress-grpc
  namespace: argocd
  annotations:
    kubernetes.io/ingress.class: "nginx"
    nginx.ingress.kubernetes.io/backend-protocol: "GRPC"
    cert-manager.io/cluster-issuer: "cloudflare-prod-issue"
spec:
  ingressClassName: nginx
  rules:
  - host: argocd-grpc.180007.xyz
    http:
      paths:
      - path: /
        pathType: Prefix
        backend:
          service:
            name: argocd-server
            port:
              number: 443
  tls:
  - hosts:
    - argocd-grpc.180007.xyz
    secretName: argocd-tls-secret-grpc
EOF
kubectl apply -f argocd-ingress-grpc.yaml

多 k8s 集群管理发布 #

合并 kubeconfig #

# ~/.kube/config
# kubectl config view --flatten --minify
# kubectl config --kubeconfig ~/.kube/config get-contexts
apiVersion: v1
clusters:
- cluster:
    certificate-authority-data: data
    server: https://192.168.100.110:6443
  name: in-cluster
- cluster:
    certificate-authority-data: data
    server: https://192.168.100.120:6443
  name: another-cluster

contexts:
- context:
    cluster: in-cluster
    user: admin
  name: in-cluster
- context:
    cluster: another-cluster
    user: admin
  name: another-cluster
# 默认访问的集群
current-context: in-cluster
kind: Config
preferences: {}
users:
- name: in-cluster
  user:
    client-certificate-data: data
    client-key-data: data
- name: another-cluster
  user:
    client-certificate-data: data
    client-key-data: data

argo添加集群 #

# 添加集群
argocd cluster add in-cluster
argocd cluster add another-cluster --kubeconfig ~/.kube/config
