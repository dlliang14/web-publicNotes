# 2. kubectl debug 命令 1.23+推荐 
kubectl debug -n test -it nginx-test-54f7774656-spqzq --image=registry.cn-shenzhen.aliyuncs.com/dll_samdeng/netshoot:v0.15 --profile=netadmin
# --profile= 选择预定义的调试配置 Options are "legacy", "general", "baseline", "netadmin", "restricted" or "sysadmin".
# restricted 安全级别较高的调试环境
# legacy 仅共享网络 简单的 ping, curl, dig 测试
# general 共享进程命名空间 最常用。可以执行 top, ps 查看业务进程
# netadmin	获得网络管理权限 需要修改路由表、执行 iptables 或高级抓包
# sysadmin 获得系统管理权限 调试内核参数、挂载文件系统（权限最高，需谨慎）

# 3. 测试抓包
curl -sLk https://test1.180007.xyz
# 直接查看
tcpdump -i any -p -nnvvv -A -s 0 port 80
# 保存到文件 不需要-vvv -A ascii输出详细内容
tcpdump -i any -p -nn -s 0 -w /tmp/capture.pcap port 80
kubectl cp -n test -c debugger-k2grp nginx-test-54f7774656-spqzq:/tmp/capture.pcap /tmp/capture.pcap

# 4. 退出后，容器会exited状态保存 资源不占用，只是yaml还在
